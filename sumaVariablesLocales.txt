#include <stdio.h>
#include <stdint.h>


int16_t x=5;

int main()
{
    int16_t a[3]={1,2,3};
    
    a[1]=x;

    printf("a[1] = %d\n",a[1]);
    
    return 0;
}

------------------------------------------------------------.data

    x: .word   5
    fmt: .string "a[1] = %d\n"

.text
.global main
main:
	# your code goes here
	
    pushq   %rbp
    movq    %rsp, %rbp
    subq    $16, %rsp
    
    //int16_t a[3]={1,2,3};
    movw    $3, -2(%rbp)                //a[2] <-- 3    ;a[2] at address: rbp-2 (highest address)
    movw    $2, -4(%rbp)                //a[1] <-- 2    ;a[1] at address: rbp-4
    movw    $1, -6(%rbp)                //a[0] <-- 1    ;a[0] at address: rbp-6 (lowest address)
    
    //a[1]=x;
    movw    x,%ax                       //ax <-- (x)
    movw    %ax,-4(%rbp)                //a[1] <-- (x)
    
    //printf("a[1] = %d\n",a[1]);
    movswl  -4(%rbp), %esi              //esi <-- (a[1])
    movl    $fmt, %edi
    movl    $0, %eax
    call    printf
    
    xorl    %eax, %eax
    addq    $16,%rsp
    popq    %rbp
	ret
